Constructor

Parameters

NameTypeDescription

(defined in /feel/node_modules/mongodb/node_modules/mongodb-core/node_modules/bson/browser_build/bson.js:)
function DBRef(namespace, oid, db) {
  if(!(this instanceof DBRef)) return new DBRef(namespace, oid, db);
  
  this._bsontype = 'DBRef';
  this.namespace = namespace;
  this.oid = oid;
  this.db = db;
};
